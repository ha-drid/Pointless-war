A file with the extension *. voxelModel works with opengl. 
It uses the instansting technology, that is, one object, for example, a cube, is drawn many times. 
To do this, the glDrawElementsInstanced or glDrawArraysInstanced function is called

example inefficient:

// begin
*.vert
#version 330

attribute vec3 position;
attribute vec3 color;

out vec4 pass_color;

void main()
{	
	gl_Position = vec4(position, 1);
	pass_color = vec4(color, 1);
}

*.frag
#version 330

in vec4 pass_color;
out vec4 color;

void main()
{	
	color = pass_color;
}
// end

example effectively:
// begin

*.vert

#version 330

attribute vec3 position;
attribute vec3 color;

uniform vec3 instance_position[100];
uniform vec3 instance_color[100];

out vec4 pass_color;

void main()
{	
	gl_Position = vec4(position + instance_position[gl_InstanceID], 1);
	pass_color = vec4(color + instance_color[gl_InstanceID], 1);
}

*.frag

#version 330

in vec4 pass_color;
out vec4 color;

void main()
{	
	color = pass_color;
}
// end
